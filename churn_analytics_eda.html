sql_query = """
            select name
            from sqlite_master
            where type = 'table';
"""

tables = pd.read_sql(sql_query, conn) #here it will gather the data from the source that we have mentioned and the connection from where we have to pick the values

#creation of dataframe for each table.
for table_name in tables['name']:
    df = pd.read_sql(f"SELECT * FROM {table_name}", conn)
    globals()[f"df_{table_name}"] = df
    print(f"Created dataframe: df_{table_name}")

conn.close()
